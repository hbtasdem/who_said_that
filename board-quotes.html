One-time Setup · ~15 mins
Connect to Google Sheets
This app uses a Google Apps Script as a tiny backend. Free, no server, no database.
Step 1 · Sheet
Step 2 · Script
Step 3 · Configure
  • 1 Go to sheets.google.com and create a new spreadsheet
  • 2 Name it something like "Board Quotes 2025"
  • 3 In row 1, add these headers:
    A1: timestamp   B1: quote   C1: encryptedName
  • 4 Copy the spreadsheet ID from the URL:
    https://docs.google.com/spreadsheets/d/YOUR_ID_IS_HERE/edit
  • 1 In your spreadsheet, go to Extensions → Apps Script
  • 2 Delete any existing code and paste this:
    // Replace with your actual spreadsheet ID
    const SHEET_ID = "PASTE_YOUR_SHEET_ID_HERE";

    function doPost(e) {
      const data = JSON.parse(e.postData.contents);
      const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
      sheet.appendRow([new Date(), data.quote, data.encryptedName]);
      return ContentService
        .createTextOutput(JSON.stringify({ok: true}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    function doGet(e) {
      const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
      const rows = sheet.getDataRange().getValues().slice(1);
      const data = rows.map(r => ({quote: r[1], encryptedName: r[2]}));
      return ContentService
        .createTextOutput(JSON.stringify(data))
        .setMimeType(ContentService.MimeType.JSON);
    }
  • 3 Click Deploy → New deployment
  • 4 Choose type: Web app, set "Who has access" to Anyone
  • 5 Click Deploy and copy the Web App URL
  • 1 Paste your Apps Script Web App URL here:
  • 2 Choose a secret encryption key and give it to a trusted friend (not yourself):
    ⚠️ Give this key to someone else. You shouldn't know it until the game!
  • 3 Save config and share the Submit URL with your board members
